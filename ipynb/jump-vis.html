def plot_manga(df, xlim=None, ylim=None, title=None):
    df[col].plot(xlim=xlim, ylim=ylim, rot=45)
    plt.legend(loc='center left', bbox_to_anchor=(1, 0.5))
    plt.xlabel('(, )')
    plt.ylabel('')
    if title:
        plt.title(title)
